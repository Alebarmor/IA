import math

class Punto(object):

		def __init__(self, x=0, y=0):
			self.x = x
			self.y = y

		def distancia_al_origen(self):
			return math.hypot(self.x, self.y)

		def __eq__(self, otro):
			return self.x == otro.x and self.y == otro.y

		def __str__(self):
			return "({0.x!r}, {0.y!r})".format(self)

			super().__init__(x, y)
			self.radio = radio
